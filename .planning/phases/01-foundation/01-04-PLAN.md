SAVE_FILENAME = "save.json"
BACKUP_COUNT = 3  (per D-03)

_save_dir() -> pathlib.Path:
  - Check os.environ.get("DEVMON_HOME") first (D-08)
  - If set, return pathlib.Path(that value)
  - Otherwise return pathlib.Path(platformdirs.user_data_dir("devmon", "devmon"))
  - Do NOT hardcode "~/.devmon" in production code (Pitfall 3)

save(state: GameState) -> None:
  - d = _save_dir(); d.mkdir(parents=True, exist_ok=True)
  - current = d / "save.json"
  - tmp = d / "save.json.tmp"
  - Rotate backups BACKWARDS (Pitfall 4 — must go from high index to low):
    for i in range(BACKUP_COUNT - 1, 0, -1):  # 2, 1
        src = d / f"save.bak{i}"
        dst = d / f"save.bak{i + 1}"  # bak2->bak3, bak1->bak2
        if src.exists(): os.replace(src, dst)
    if current.exists(): os.replace(current, d / "save.bak1")
  - Write: tmp.write_text(state.model_dump_json(indent=2), encoding="utf-8")
  - Atomic rename: os.replace(tmp, current)

load() -> GameState | None:
  - d = _save_dir()
  - candidates = [d / "save.json"] + [d / f"save.bak{i}" for i in range(1, BACKUP_COUNT + 1)]
  - For each path in candidates:
    - If not path.exists(): continue
    - Try: raw = json.loads(path.read_text(encoding="utf-8"))
    - Call raw = migrate(raw) from devmon.persistence.migrations
    - Call return GameState.model_validate(raw)
    - Except any Exception:
      - corrupted_path = path.with_suffix(".corrupt.bak") or path.parent / (path.name + ".corrupt.bak")
      - path.rename(corrupted_path) (D-16: keep for investigation)
      - continue
  - If all candidates exhausted: return None

  assert (tmp_save_dir / "save.json").exists()

test_schema_version(tmp_save_dir):
  import json as json_mod
  state = GameState.new_game("Ash")
  save(state)
  data = json_mod.loads((tmp_save_dir / "save.json").read_text(encoding="utf-8"))
  assert data["schema_version"] == 1

test_data_dir(tmp_save_dir):
  new_dir = tmp_save_dir / "nested" / "subdir"
  os.environ["DEVMON_HOME"] = str(new_dir)
  state = GameState.new_game("Ash")
  save(state)
  assert (new_dir / "save.json").exists()
  os.environ["DEVMON_HOME"] = str(tmp_save_dir)  # reset

test_backup_rotation(tmp_save_dir):
  for name in ["A", "B", "C", "D"]:
      save(GameState.new_game(name))
