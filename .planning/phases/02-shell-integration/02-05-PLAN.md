<!-- Pattern 4 from RESEARCH.md — event processing flow -->
Process on every devmon invocation:
1. load config
2. resolve event log path from config["shell"]["event_log"]
3. read_and_consume(log_path) -> events
4. if events: load state, process_events(state, events, config), save(state)
5. Continue with whatever subcommand was invoked

    Add the private helper function before `app.callback()`:

    ```python
    def _process_event_log_on_startup() -> None:
        """Process accumulated shell events and award XP on every devmon invocation.

        Pattern 4 (RESEARCH.md): Read → process → save → continue.
        Fails silently — game activity must never block normal devmon usage.
        """
        try:
            config = load_config()
        except Exception:
            config = DEFAULT_CONFIG

        try:
            log_path = Path(config.get("shell", {}).get("event_log", DEFAULT_CONFIG["shell"]["event_log"]))
            events = read_and_consume(log_path)
            if not events:
                return  # Nothing to process — fast path

            state = load_state()
            if state is None:
                # No save exists yet — create one so XP has somewhere to go
                from devmon.models.state import GameState
                state = GameState.new_game("Player")

            process_events(state, events, config)
            save_state(state)
        except Exception:
            pass  # Never block the user's terminal workflow
    ```

    Modify the `@app.callback()` function to call the helper. Insert `_process_event_log_on_startup()` call at the START of the `main()` function body (before the version flag logic — the version flag check raises typer.Exit() so it must come after):

    ```python
    # ---------------------------------------------------------------------------
    # devmon track — explicit activity tracking commands (TRACK-03, Pattern 6)
    # ---------------------------------------------------------------------------

    track_app = typer.Typer(
        name="track",
        help="Manually track coding events for XP rewards.",
        no_args_is_help=True,
    )


    @track_app.command("test-pass")
    def track_test_pass() -> None:
        """Record a test suite pass event for bonus XP.

        Add to your test alias: alias pytest='pytest && devmon track test-pass'
        XP is awarded on the next devmon invocation. (TRACK-03)
        """
        import json
        import time
        import os

        try:
            from devmon.config.loader import load_config
            config = load_config()
        except Exception:
            from devmon.config.defaults import DEFAULT_CONFIG
            config = DEFAULT_CONFIG

        log_path = Path(config.get("shell", {}).get(
            "event_log",
            config.get("shell", {}).get("event_log", str(Path.home() / ".local" / "share" / "devmon" / "devmon" / "events.log"))
        ))
        log_path.parent.mkdir(parents=True, exist_ok=True)

        event = {
            "ts": int(time.time() * 1000),
            "exit": 0,
            "dur": 0,
            "cwd": os.getcwd(),
            "type": "test_pass",
        }

        with log_path.open("a", encoding="utf-8") as f:
            f.write(json.dumps(event) + "\n")

        console.print("[green]Test pass recorded![/green] XP will be awarded on next devmon invocation.")
    ```
